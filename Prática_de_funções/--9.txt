void main() {
  compara(15,12);
}

void compara(int pri, int seg) {
  
  if(pri<seg){

    for(int i = pri; i <= seg; i++){
      
      print ("$i");
    }
    
  }
  else if (seg<pri){
    
    for(int i = pri; i >= seg; i--){
      
      print ("$i");
    }
   
  }
  else
    print("Valores Iguais");
}
